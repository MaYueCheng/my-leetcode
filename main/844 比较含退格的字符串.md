# [844 比较含退格的字符串](https://leetcode.cn/problems/backspace-string-compare/)

## 双指针

```c++
bool backspaceCompare(string s, string t) {
    int fast_s = 0;
    int slow_s = 0;
    for(; fast_s < s.size(); ++fast_s){
        if(s[fast_s] == '#'){
            slow_s > 0 ? --slow_s: slow_s = 0;
        }else{
            s[slow_s++] = s[fast_s];
        }
    }

    int fast_t = 0;
    int slow_t = 0;
    for(; fast_t < t.size(); ++fast_t){
        if(t[fast_t] == '#'){
            slow_t > 0 ? --slow_t: slow_t = 0;
        }else{
            t[slow_t++] = t[fast_t];
        }
    }

    if(s.substr(0, slow_s) != t.substr(0, slow_t)){
        return false;
    }
    return true;
}
```

## 双指针+栈思想

```c++
bool backspaceCompare(string s, string t) {
    std::string s1 = {};
    std::string t1 = {};
    for(const auto &character: s){
        if(character != '#'){
            s1.push_back(character);
        }else if(!s1.empty()){
            s1.pop_back();
        }
    }
    std::cout << s1 << std::endl;
    for(const auto &character: t){
        if(character != '#'){
            std::cout << character << std::endl;
            t1.push_back(character);
        }else if(!t1.empty()){
            t1.pop_back();
        }
    }
    return (s1 == t1) ? true : false;
}
```
