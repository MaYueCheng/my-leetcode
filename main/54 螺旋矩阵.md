vector<int> spiralOrder(vector<vector<int>>& matrix) {
    int n = matrix.size();
    int m = matrix[0].size();
    vector<int> ret;
    ret.reserve(n * m);
    int top = 0;
    int right = m - 1;
    int bottom = n - 1;
    int left = 0;
    int i = 0;
    int j = 0;
    while(left <= right && top <= bottom) {
        i = top;
        j = left;
        // top
        for(; j < right; ++j) {
            ret.push_back(matrix[top][j]);
        }
        // right
        for(; i < bottom; ++i) {
            ret.push_back(matrix[i][right]);
        }
        // bottom
        for(; j > left; --j) {
            ret.push_back(matrix[bottom][j]);
        }
        // left
        for(; i > top; --i) {
            ret.push_back(matrix[i][left]);
        }
        
        ++top;
        --right;
